## Install

```bash
pnpm add @florexlabs/ui
pnpm add react react-dom tailwindcss
pnpm add @phosphor-icons/react  # optional
```

## Setup

Import the CSS tokens in your app entry:

```css
@import "tailwindcss";
@import "@florexlabs/ui/florex.css";
```

## Quick Start

```tsx
import { Button, Card, Badge } from "@florexlabs/ui";

<Card variant="hover">
  <Button variant="primary">Save</Button>
  <Badge tone="brand">New</Badge>
</Card>
```
